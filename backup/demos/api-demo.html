const mbti = new MBTIClient('your-api-key');

const result = await mbti.analyze({
    answers: [
        { dimension: 'EI', value: 'E', weight: 2 },
        { dimension: 'SN', value: 'N', weight: 1 }
    ]
});

console.log(result.type); // "ENTJ"
console.log(result.title); // "지휘관"
# MBTI 분석 API 사용 예시
import mbti_client

client = mbti_client.Client('your-api-key')

result = client.analyze({
    'answers': [
        {'dimension': 'EI', 'value': 'E', 'weight': 2}
    ]
})

print(result['type'])  # "ENTJ"
# MBTI 분석 API 사용 예시

  -H "Content-Type: application/json" \
  -d '{
    "answers": [
        {"dimension": "EI", "value": "E", "weight": 2}
    ]
  }'
